#version 330 core

in vec2 fragment_tex_coord;
in vec3 fragment_normal;
in vec3 fragment_world_pos;

out vec4 screen_color;

uniform sampler2D my_texture;
uniform vec3 light_direction;
uniform vec3 light_color;
uniform float k;

void main(){
    vec3 reflectivity = texture(my_texture, fragment_tex_coord).rgb;
    vec3 norm = normalize(fragment_normal);
    vec3 lightDir = normalize(-light_direction);
    
    vec3 ambient = k * light_color;
    float diff = max(dot(norm, lightDir), 0.0);
    vec3 diffuse = diff * light_color;
    
    vec3 result = (ambient + diffuse) * reflectivity;
    screen_color = vec4(result, 1.0);
}
